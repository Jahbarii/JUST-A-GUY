// Cookie management  
function setCookie(name, value, days) {  
const expires = new Date(Date.now() + days * 864e5).toUTCString();  
document.cookie = name + '=' + encodeURIComponent(value) + '; expires=' +  
expires + '; path=/; SameSite=Lax';  
}
function getCookie(name) {  
return document.cookie.split('; ').reduce((r, v) => {  
const parts = v.split('=');  
return parts[0] === name ? decodeURIComponent(parts[1]) : r;  
}, '');  
}
function showCookieBanner() {  
const consent = getCookie('cookie-consent');  
if (!consent) {  
document.getElementById('cookie-banner').classList.add('active');  
} else {  
updateConsentDisplay(consent);  
}
}
function acceptCookies() {  
setCookie('cookie-consent', 'accepted', 365);  
document.getElementById('cookie-banner').classList.remove('active');  
updateConsentDisplay('accepted');  
// Ici tu peux activer Google Analytics ou autre  
}
function refuseCookies() {  
setCookie('cookie-consent', 'refused', 365);  
document.getElementById('cookie-banner').classList.remove('active');  
updateConsentDisplay('refus');  
}
function updateConsentDisplay(status) {  
const display = document.getElementById('current-consent');  
if (display) {  
display.textContent = status === 'accepted' ? 'Tous les cookies accepts' :  
'Cookies essentiels uniquement';  
}
}
